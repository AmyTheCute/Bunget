cmake_minimum_required(VERSION 3.5)

project(Bunget VERSION 1.0
DESCRIPTION "Budget Bunny"
LANGUAGES CXX)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_INCLUDE_CURRENT_DIR ON)
set(CMAKE_PREFIX_PATH "/Users/amyhaghighimood/Qt/lib/cmake")

find_package(QT NAMES Qt6 Qt5 REQUIRED COMPONENTS Widgets) 
find_package(Qt${QT_VERSION_MAJOR} REQUIRED COMPONENTS Widgets)

file(GLOB PROJECT_SOURCES
     "src/GUI/*.h"
     "src/GUI/*.cpp"
     "src/GUI/*.ui"
)

set (CMAKE_CXX_STANDARD 20)


add_subdirectory(src/Financial)

add_executable(Bunget ${PROJECT_SOURCES})

target_link_libraries(Bunget PUBLIC Financial)
target_link_libraries(Bunget PRIVATE Qt${QT_VERSION_MAJOR}::Widgets)
